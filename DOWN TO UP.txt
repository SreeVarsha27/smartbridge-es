#ifndef F_CPU
#define F_CPU 16000000UL
#endif
#include <avr/io.h>
#include <util/delay.h>


int main(void)
{
   DDRC=0XFF;
       while (1) 
    {
	PORTC=0b00000001;
	_delay_ms(1000);
	PORTC=0b00000010;
	_delay_ms(1000);
	PORTC=0b00000100;
	_delay_ms(1000);
	PORTC=0b00001000;
	_delay_ms(1000);
	PORTC=0b00010000;
	_delay_ms(1000);
	PORTC=0b00100000;
	_delay_ms(1000);
	PORTC=0b010000000;
	_delay_ms(1000);
	PORTC=0b100000000;
	_delay_ms(1000);
		
    }
}
