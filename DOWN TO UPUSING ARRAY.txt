#ifndef F_CPU
#define F_CPU 16000000UL
#endif
#include <avr/io.h>
#include<util/delay.h>

int main(void)
{
	
	int i;
	

int a[8]={0b00000001,0b00000010,0b00000100,0b00001000,0b00010000,0b00100000,0b01000000,0b10000000};

	DDRC=0XFF;
    
    while (1) 
    {
		for(i=1;i<=8;i++)
		{
		PORTC=a[i];
		_delay_ms(1000);	
		}
		
		
    }
}
