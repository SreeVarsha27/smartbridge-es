#define F_CPU 16000000UL
#include "avr/io.h"
#include <util/delay.h>
void PWM_init()
{
	TCCR0=(1<<WGM00)|(1<<WGM01)|(1<<COM01)|(1<<CS00);
	DDRB=(1<<PB3);
}

int main()
{
	unsigned char duty;
	PWM_init();
	DDRC=0X00;

	
	while (1)
	{
		if((PINC&(1<<PC0))==0)
		{
			duty=200;
			
			OCR0=duty;
			_delay_ms(8);
			
		}
		if((PINC&(1<<PC1))==0)
		{
			duty=150;
			OCR0=duty;
			_delay_ms(8);
		}
		
	}
}
