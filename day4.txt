		   {

3)write a program for LCD

/*
 * lcd.c
 *
 * Created: 07-06-2019 12:30:23
 * Author : SREE VARSHA
 */ 

#include <avr/io.h>
#define F_CPU 16000000UL
#include <util/delay.h>

#define RS 7
#define E 5
 
void send_command(unsigned char command);
void send_character(unsigned char character);

int main(void)
{
	DDRD=0xFF;
	DDRC=0xFF;
	_delay_ms(50);
	send_command(0x01);
	send_command(0x38);
	send_command(0x0F);
	send_character(0x65);
	send_character(0x66);
	send_character(0x78);
	
}
void send_command(unsigned char command)
{
	PORTC=command;
	PORTD&=~(1<<RS);
	PORTD|=(1<<E);
	_delay_ms(50);
	PORTD&=~(1<<E);
	PORTC=0;
}
void send_character(unsigned char character)
{
	PORTC=character;
	PORTD|=(1<<RS);
	PORTD|=(1<<E);
	_delay_ms(50);
	PORTD&=~(1<<E);
	PORTC=0;
}
