#include <avr/io.h>
#define F_CPU 16000000UL
#include <util/delay.h>
#include <stdlib.h>
#define E 5
#define RS 7
void send_command(unsigned char command);
void send_character(unsigned char character);
void send_string(char *string);


int main(void)
{
    DDRC=0xFF;
	DDRA=0x00;
	DDRD=0xFF;
	DDRB&=~(1<<PB0);

	_delay_ms(50);
	ADMUX|=(1<<REFS0)|(1<<REFS1);
	ADCSRA|=(1<<ADEN)|(1<<ADATE)|(1<<ADPS0)|(1<<ADPS1)|(1<<ADPS2);
	int16_t counta=0;
	char s[3];
	
	send_command(0x01);
	_delay_ms(50);
	send_command(0x38);
	_delay_ms(50);
	send_command(0x0F);
	_delay_ms(50);
	ADCSRA|=(1<<ADSC);
    while (1) 
    {
		
		if((PINB & (1<<PB0))==0)
		{
			
		
		counta=ADC/4;
		send_string("smartbridge");
		send_command(0x80+0x40+0);
		send_string("temp(c)=");
		send_command(0x80+0x40+8);
		itoa(counta,s,10);
		send_string(s);
		send_string("  ");
		send_command(0x80+0);
		send_command(0x01);
		_delay_ms(10000);
		}
		
		
		
	}
}
void send_string(char *string)
{
	while(*string>0)
	{
		send_character(*string++);
	}
}
void send_command(unsigned char command)
{
	PORTC=command;
	PORTD&=~(1<<RS);
	PORTD|=(1<<E);
	_delay_ms(50);
	PORTD&=~(1<<E);
	PORTC=0;
}
void send_character(unsigned char character)
{
	PORTC=character;
	PORTD|=(1<<RS);
	PORTD|=(1<<E);
	_delay_ms(50);
	PORTD&=~(1<<E);
	PORTC=0;
}
